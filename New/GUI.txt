    private void btnProcessActionPerformed(java.awt.event.ActionEvent evt) {                                           

        double price = 0.0;
        int qty = 0;
        String prodCategory = "";

        if (!radWM.isSelected() && !radNS.isSelected()) {
            JOptionPane.showMessageDialog(null, "Please select a product category.");
            return;
        }

        if (radWM.isSelected()) {
            prodCategory = radWM.getText();
        } else if (radNS.isSelected()) {
            prodCategory = radNS.getText();
        }

        try {
            price = Double.parseDouble(txtPrice.getText());
            qty = Integer.parseInt(txtQty.getText());
            int prodCode = Integer.parseInt(txtProdCode.getText());
            ProductPD prod = new ProductPD(txtProdName.getText(), price, qty, prodCode, ProductPD.ProdCategory.valueOf(prodCategory));
            ProductPD.addProduct(prod);
            clearControls();
            JOptionPane.showMessageDialog(null, "Product added successfully.");
        } catch (NumberFormatException ex) {
            JOptionPane.showMessageDialog(null, "Please enter valid numeric values for price, quantity, and product code.");
        } catch (IllegalArgumentException ex) {
            JOptionPane.showMessageDialog(null, ex.getMessage());
        }
    }                                          

    private void btnClearActionPerformed(java.awt.event.ActionEvent evt) {                                         
        clearControls();
        radWM.setSelected(false);
        radNS.setSelected(false);
    }                                        

    private void btnNextProductActionPerformed(java.awt.event.ActionEvent evt) {                                               

        if (k < items.size() - 1) {
            k++;
            displayOnTextFields();
        } else {
            JOptionPane.showMessageDialog(null, "Reached the end of the list.");
        }

    }                                              

    private void btnPreviousActionPerformed(java.awt.event.ActionEvent evt) {                                            

        if (k > 0) {
            k--;
            displayOnTextFields();
        } else {
            JOptionPane.showMessageDialog(null, "Reached the beginning of the list.");
        }


    }                                           

    private void btnAddButtonActionPerformed(java.awt.event.ActionEvent evt) {                                             
        pnlAddProduct.setVisible(true);
        clearControls();
        txtProdCode.setEditable(false);
        txtStockValue.setEditable(false);

    }                                            

    private void btnOutofStockActionPerformed(java.awt.event.ActionEvent evt) {                                              
        ArrayList<ProductPD> outOfStockItems = ProductPD.viewOutofStockItems();
        String displayText = "";
        for (ProductPD prod : outOfStockItems) {
            displayText += prod.toString() + "\n";
        }
        taDisplay.setText(displayText);
    }                                             

    private void btnReplenishActionPerformed(java.awt.event.ActionEvent evt) {                                             
        try {
            int productCode = Integer.parseInt(JOptionPane.showInputDialog("Enter product code:"));
            int quantity = Integer.parseInt(JOptionPane.showInputDialog("Enter quantity:"));
            ProductPD.replenish(quantity, productCode);
            JOptionPane.showMessageDialog(null, "Product replenished successfully.");
        } catch (NumberFormatException ex) {
            JOptionPane.showMessageDialog(null, "Please enter valid numeric values.");
        }
    }                                            

    private void groupRadioButtons() {

        ButtonGroup radButtonGrp = new ButtonGroup();
        radButtonGrp.add(radWM);
        radButtonGrp.add(radNS);
    }

    private void displayOnTextFields() {
        NumberFormat currencyFormat = NumberFormat.getCurrencyInstance();
        if (items != null && !items.isEmpty() && k < items.size()) {
            ProductPD prod = items.get(k);
            txtProdCode.setText(String.valueOf(prod.getProdCode()));
            txtProdName.setText(prod.getProdName());
            txtPrice.setText(currencyFormat.format(prod.getPrice()));
            txtQty.setText(String.valueOf(prod.getQuantity()));
            txtStockValue.setText(currencyFormat.format(prod.getPrice() * prod.getQuantity()));
        }

    }

    private void clearControls() {
        txtProdCode.setText("");
        txtProdName.setText("");
        txtPrice.setText("");
        txtQty.setText("");
        txtStockValue.setText("");
        radNS.setSelected(false);
        radWM.setSelected(false);
        txtProdName.requestFocus();
    }
